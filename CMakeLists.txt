set(CMAKE_DEBUG_POSTFIX d)
set(CMAKE_WINDOWS_EXPORT_ALL_SYMBOLS FALSE CACHE BOOL "Export all symbols")
option(BUILD_SHARED_LIBS "Build shared libraries (.dll/.so) instead of static ones (.lib/.a)" ON)

ADD_DEFINITIONS(-D OPENNURBS_EXPORTS)
add_library(opennurbs_public ${SOURCES})

if(WIN32)


target_link_libraries(opennurbs_public shlwapi.lib)
endif()

set_target_properties(opennurbs_public PROPERTIES
    VERSION ${PROJECT_VERSION}
    SOVERSION 1
    PUBLIC_HEADER "${HEADERS}")
target_include_directories(opennurbs_public
    PUBLIC
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
        $<INSTALL_INTERFACE:${CMAKE_INSTALL_INCLUDEDIR}>
    PRIVATE
        .
)
	  
set ( opennurbs_public_INCLUDE_DIRS "${CMAKE_CURRENT_SOURCE_DIR}" )

# Destination of the installed config files (relative path):
#set ( CMAKE_CONFIG_DEST "share/cmake/Modules" )

# We configure our template. The template is described later.
configure_package_config_file (                         
        "opennurbs_public-config.cmake.in"
        # Important to write in CMAKE_BINARY_DIR if you want the registry
        # mechanism to work:
        "${CMAKE_CURRENT_BINARY_DIR}/opennurbs_public-config.cmake"  
        INSTALL_DESTINATION "${CMAKE_INSTALL_DATADIR}/lib/cmake"
        PATH_VARS opennurbs_public_INCLUDE_DIRS )

# This file is included in our template:
export ( TARGETS opennurbs_public FILE "${CMAKE_CURRENT_BINARY_DIR}/opennurbs_publicTargets.cmake" )

export ( PACKAGE opennurbs_public ) 


install(TARGETS opennurbs_public
    EXPORT opennurbs_publicTargets
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    PUBLIC_HEADER DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}
         ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR}
         RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR})

	
install(
    EXPORT opennurbs_publicTargets 
    FILE opennurbs_publicTargets.cmake
    DESTINATION "${CMAKE_INSTALL_DATADIR}/cmake"
)

install(
    FILES "${CMAKE_CURRENT_BINARY_DIR}/opennurbs_public-config.cmake"
    DESTINATION "${CMAKE_INSTALL_DATADIR}/cmake"
)
